\subsection{À la charge}
\begin{frame}
	\frametitle{Mises en équation de la charge}
	\begin{align}
		V_\mathrm{in}(0) & = 12; \quad
		V_C(0)             = -12; \quad
		V_D(0)             = 0.6; \quad
		V_R(0)             = \SI{23.4}{\V}                               \\
		V_\mathrm{in}(t) & = V_C(t) + V_D(t) + V_R(t) \Rightarrow
		V_\mathrm{in}^{\prime}(t) = \frac{1}{RC}V_R(t) + V_R^{\prime}(t) \\
		V_R(t)           & = 23.4\e^{\frac{-t}{1.5\times10^{-5}}}
	\end{align}
\end{frame}
